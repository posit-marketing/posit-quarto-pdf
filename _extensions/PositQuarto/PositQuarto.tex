%% Define extension path, using repo name and extension name

\newcommand{\reponame}{posit-marketing}

  \newcommand{\extensionpath}{_extensions/\reponame/\extensionname}

  \newcommand{\extensionpath}{_extensions/\extensionname}
}

%% Define the header and footer image file
\newcommand{\headerimage}{\extensionpath/header.png}
\newcommand{\footerimage}{\extensionpath/footer.png}


%% Let's add a border at the top
\AddToShipoutPicture{%
    % Image at the top
    \AtPageLowerLeft{%
        \put(0,\LenToUnit{\dimexpr\paperheight-2.7cm}){%
            \color{light}\makebox[\paperwidth]{\includegraphics[width=\paperwidth,height=3cm,keepaspectratio]{\headerimage}%
        }%
    }%
}%
}

%% Style the pages
\fancypagestyle{default}{
  \fancyhf{}
  \renewcommand\headrulewidth{0pt}
  \fancyfoot[R]{\thepage}
  \fancyfootoffset{3.5cm}
}
\setlength{\footskip}{20pt}

%% Style the last page
\fancypagestyle{last-page}{
  \fancyhf{}
  \renewcommand\headrulewidth{0pt}
  \fancyfoot[R]{\thepage}
  \fancyfoot[C]{\includegraphics[width=\paperwidth]{\footerimage}} % Add image to the footer
  \fancyfootoffset{3.5cm}
}

%% style the chapter/section fonts

%% Define font path
\newcommand{\fontpath}{\extensionpath/Open_Sans/static/}

%% Use some custom fonts
\setsansfont{OpenSans}[
    Path=\fontpath,
    Scale=0.9,
    Extension = .ttf,
    UprightFont=*-Regular,
    BoldFont=*-Bold,
    ItalicFont=*-Italic,
    ]

\setmainfont{OpenSans}[
    Path=\fontpath,
    Scale=1.1,
    Extension = .ttf,
    UprightFont=*-Regular,
    BoldFont=*-Bold,
    ItalicFont=*-Italic,
    ]
    
%% Set the font for headings
\newfontfamily\headingfont{OpenSans}[
    Path=\fontpath,
    Extension = .ttf,
    UprightFont=*-Light,
    BoldFont=*-Light,
    ItalicFont=*-Light
]

\titleformat*{\section}{\LARGE\headingfont}
\titleformat*{\subsection}{\Large\headingfont}
\titleformat*{\subsubsection}{\large\headingfont}

}
